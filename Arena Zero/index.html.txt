// ── AUDIO ──────────────────────────────────────────────────────────
let AC = null;
let musicStep = 0, musicNext = 0, musicTmr = null;
const BPM = 115, BEAT = 60 / BPM;
const BASS_P = [110,0,110,0,98,0,110,0, 110,0,0,110,82,0,98,0];
const LEAD_P = [0,0,220,0, 0,261,0,0, 0,0,196,0, 0,247,0,220];

function initAudio() {
  if (AC) return;
  AC = new (window.AudioContext || window.webkitAudioContext)();
  musicNext = AC.currentTime + 0.1;
  musicStep = 0;
  musicTick();
}

function noiseBuf(dur) {
  const n = Math.ceil(AC.sampleRate * dur);
  const buf = AC.createBuffer(1, n, AC.sampleRate);
  const d = buf.getChannelData(0);
  for (let i = 0; i < n; i++) d[i] = (Math.random() * 2 - 1) * (1 - i / n);
  return buf;
}

function musicTick() {
  while (musicNext < AC.currentTime + 0.15) {
    const s = musicStep % BASS_P.length, t = musicNext;
    if (BASS_P[s]) {
      const o = AC.createOscillator(), g = AC.createGain();
      o.type = 'sawtooth'; o.frequency.value = BASS_P[s];
      g.gain.setValueAtTime(0.2, t); g.gain.exponentialRampToValueAtTime(0.001, t + BEAT * 0.85);
      o.connect(g); g.connect(AC.destination); o.start(t); o.stop(t + BEAT);
    }
    if (s % 4 === 0) {
      const o = AC.createOscillator(), g = AC.createGain();
      o.type = 'sine'; o.frequency.setValueAtTime(170, t); o.frequency.exponentialRampToValueAtTime(32, t + 0.16);
      g.gain.setValueAtTime(0.5, t); g.gain.exponentialRampToValueAtTime(0.001, t + 0.18);
      o.connect(g); g.connect(AC.destination); o.start(t); o.stop(t + 0.2);
    }
    if (s === 4 || s === 12) {
      const src = AC.createBufferSource(), g = AC.createGain(), f = AC.createBiquadFilter();
      src.buffer = noiseBuf(0.12); f.type = 'bandpass'; f.frequency.value = 1800; f.Q.value = 0.6;
      g.gain.setValueAtTime(0.22, t); g.gain.exponentialRampToValueAtTime(0.001, t + 0.12);
      src.connect(f); f.connect(g); g.connect(AC.destination); src.start(t);
    }
    if (s % 2 === 0) {
      const src = AC.createBufferSource(), g = AC.createGain(), f = AC.createBiquadFilter();
      src.buffer = noiseBuf(0.04); f.type = 'highpass'; f.frequency.value = 6500;
      g.gain.setValueAtTime(s % 4 === 2 ? 0.05 : 0.1, t); g.gain.exponentialRampToValueAtTime(0.001, t + 0.04);
      src.connect(f); f.connect(g); g.connect(AC.destination); src.start(t);
    }
    if (LEAD_P[s]) {
      const o = AC.createOscillator(), o2 = AC.createOscillator(), g = AC.createGain();
      o.type = 'square'; o2.type = 'square';
      o.frequency.value = LEAD_P[s]; o2.frequency.value = LEAD_P[s] * 1.007;
      g.gain.setValueAtTime(0.06, t); g.gain.exponentialRampToValueAtTime(0.001, t + BEAT * 0.75);
      o.connect(g); o2.connect(g); g.connect(AC.destination);
      o.start(t); o2.start(t); o.stop(t + BEAT); o2.stop(t + BEAT);
    }
    musicStep++; musicNext += BEAT / 2;
  }
  musicTmr = setTimeout(musicTick, 22);
}

function addW(x, y, w, h)     { mapObjs.push({ x, y, w, h, type: 'wall',   solid: true, hp: -1 }); }

function addFence(x, y, w, h)  { mapObjs.push({ x, y, w, h, type: 'fence',  solid: true, hp: -1 }); }

function pushOut(e) {
  for (const o of mapObjs) {
    if (!o.solid) continue;
    const ocx = o.x + o.w / 2, ocy = o.y + o.h / 2;
    const dx = e.x - ocx, dy = e.y - ocy;
    const ox = e.w / 2 + o.w / 2 - Math.abs(dx);
    const oy = e.h / 2 + o.h / 2 - Math.abs(dy);
    if (ox > 0 && oy > 0) { if (ox < oy) e.x += dx > 0 ? ox : -ox; else e.y += dy > 0 ? oy : -oy; }
  }
}

function bulletHitsMap(bx, by, fromPlayer) {
  for (let i = 0; i < mapObjs.length; i++) {
    const o = mapObjs[i];
    if (!o.solid) continue;
    if (bx > o.x && bx < o.x + o.w && by > o.y && by < o.y + o.h) {
      if (fromPlayer && o.type === 'crate' && o.hp > 0) {
        o.hp -= 25;
        spawnPfx(bx, by, 5, '#aa7733');
        if (o.hp <= 0) { o.solid = false; spawnPfx(o.x + o.w / 2, o.y + o.h / 2, 14, '#aa7733'); }
      } else { spawnPfx(bx, by, 3, '#888888'); }
      return true;
    }
  }
  return false;
}

// ── WAVE / ENEMY CONFIG ────────────────────────────────────────────
const isBossWave = w => w % 2 === 0;

function getTypes(w) {
  if (w <= 1) return ['basic'];
  if (w === 2) return ['basic','fast','fast'];
  if (w === 3) return ['basic','fast','ranged'];
  if (w === 4) return ['basic','fast','ranged','shield'];
  return ['basic','fast','ranged','shield','heavy'];
}

function getStats(type, w) {
  const n = Math.ceil(w / 2);
  const s = {
    basic:  { hp:30+n*10,  spd:0.9+n*0.10,  dmg:10, pts:10, cn:8  },
    fast:   { hp:18+n*6,   spd:1.6+n*0.14,  dmg:8,  pts:8,  cn:6  },
    ranged: { hp:40+n*12,  spd:0.75+n*0.08, dmg:12, pts:15, cn:10 },
    shield: { hp:50+n*14,  spd:0.85+n*0.09, dmg:15, pts:20, cn:12 },
    heavy:  { hp:75+n*20,  spd:0.45+n*0.06, dmg:22, pts:50, cn:20 },
  }[type] || { hp:30, spd:1, dmg:10, pts:10, cn:8 };
  return { ...s, maxHp: s.hp };
}

function waveCfg(w) {
  const n = Math.ceil(w / 2);
  return { count: 5+(n-1)*3, iv: Math.max(380, 2000-n*80) };
}

function startWave(w) {
  boss = null; bossProjs = []; aoeZones = []; shockwaves = []; enemies = []; enemyBullets = [];
  if (isBossWave(w)) {
    toSpawn = 0; waveLeft = 1; spawnBoss(w);
  } else {
    const c = waveCfg(w); toSpawn = c.count; waveLeft = c.count; spawnIv = c.iv; spawnTimer = 0;
  }
}

// ── SPAWN ──────────────────────────────────────────────────────────
function getSpawnPos() {
  const m = WT + 55, side = Math.floor(Math.random() * 4);
  if (side === 0) return { x: m + Math.random()*(WW-m*2), y: m };
  if (side === 1) return { x: WW-m, y: m + Math.random()*(WH-m*2) };
  if (side === 2) return { x: m + Math.random()*(WW-m*2), y: WH-m };
  return { x: m, y: m + Math.random()*(WH-m*2) };
}

function spawnEnemy() {
  const pos = getSpawnPos();
  const types = getTypes(wave);
  const type = types[Math.floor(Math.random() * types.length)];
  const stats = getStats(type, wave);
  const wh = { basic:[14,16], fast:[12,14], ranged:[14,18], shield:[16,18], heavy:[18,20] }[type] || [14,16];
  const e = { x:pos.x, y:pos.y, w:wh[0], h:wh[1], type, ...stats, dmgTimer:0, anim:Math.random()*Math.PI*2, hurtFlash:0 };
  if (type === 'fast')   { e.zigTimer = 0.3; e.zigDir = 1; }
  if (type === 'ranged') { e.shootTimer = 1.5 + Math.random(); }
  if (type === 'shield') { e.shieldAngle = 0; }
  enemies.push(e);
}

function spawnPfx(x, y, n, color) {
  for (let i = 0; i < n; i++) {
    const a = Math.random()*Math.PI*2, s = 40+Math.random()*130, l = 0.35+Math.random()*0.45;
    particles.push({ x, y, vx:Math.cos(a)*s, vy:Math.sin(a)*s, r:2+Math.random()*3, color, alpha:1, life:l, maxLife:l });
  }
}
function spawnTxt(x, y, text, color) { floatTexts.push({ x, y, text, color, alpha:1, vy:-55, life:0.85, maxLife:0.85 }); }

function rrOvlp(ax,ay,aw,ah,bx,by,bw,bh) {
  return ax-aw/2<bx+bw/2 && ax+aw/2>bx-bw/2 && ay-ah/2<by+bh/2 && ay+ah/2>by-bh/2;
}
function crOvlp(cx,cy,cr,rx,ry,rw,rh) {
  const nx=Math.max(rx-rw/2,Math.min(cx,rx+rw/2)), ny=Math.max(ry-rh/2,Math.min(cy,ry+rh/2));
  return (cx-nx)**2+(cy-ny)**2<=cr*cr;
}
function angDiff(a, b) { return Math.abs(((a-b)%(Math.PI*2)+Math.PI*3)%(Math.PI*2)-Math.PI); }

function dmgPlayer(dmg) {
  if (player.shieldHp > 0) {
    const abs = Math.min(dmg, player.shieldHp); player.shieldHp = Math.max(0, player.shieldHp-abs); dmg -= abs;
    spawnPfx(player.x, player.y, 4, '#4488ff');
    if (dmg <= 0) { player.hitFlash = Math.max(player.hitFlash, 0.4); return; }
  }
  player.hp = Math.max(0, player.hp - dmg); player.hitFlash = 1;
  if (AC) sndHurt();
  spawnPfx(player.x, player.y, 5, '#ff4444');
  if (player.hp <= 0) { state = 'GAMEOVER'; spawnPfx(player.x, player.y, 24, '#ffaa00'); }
}

function resetGame() {
  state = 'PLAYING'; score = 0; coins = 0; wave = 1; gameOverAlpha = 0;
  Object.assign(player, { x:WW/2, y:WH/2, hp:100, maxHp:100, hitFlash:0, inv:0,
    fireCd:0, spd:180, spdLvl:0, shieldHp:0, shieldMax:0, armCount:0, armAngles:[0,Math.PI], armCds:[0,0] });
  cam.x = WW/2 - W/2; cam.y = WH/2 - H/2;
  mw.x = W/2 + cam.x; mw.y = H/2 + cam.y;
  bullets=[]; enemies=[]; particles=[]; floatTexts=[];
  bossProjs=[]; aoeZones=[]; shockwaves=[]; enemyBullets=[]; boss=null; transTimer=0;
  for (const k in keys) keys[k] = false;
  mouse.down=false; tl.active=false; tl.dx=0; tl.dy=0; tr.active=false;
  genMap(); startWave(1);
}

// ── UPDATE ─────────────────────────────────────────────────────────
function update(dt) {
  if (state === 'GAMEOVER') { gameOverAlpha = Math.min(1, gameOverAlpha+dt*1.5); updPfx(dt); return; }
  if (state === 'SHOP') { updPfx(dt); updFtxt(dt); return; }
  if (state === 'WAVE_TRANSITION') {
    transTimer -= dt;
    if (transTimer <= 0) { state = 'SHOP'; sndShopOpen(); }
    updPfx(dt); return;
  }
  updCam(dt); updPlayer(dt); updArms(dt);
  updBullets(dt); updEBullets(dt);
  updSpawn(dt); updEnemies(dt);
  updBoss(dt); updBossProjs(dt); updAoe(dt); updShock(dt);
  updPfx(dt); updFtxt(dt);
}

function updCam(dt) {
  const tx = player.x - W/2, ty = player.y - H/2;
  cam.x += (tx - cam.x) * Math.min(1, dt*9);
  cam.y += (ty - cam.y) * Math.min(1, dt*9);
  cam.x = Math.max(0, Math.min(WW-W, cam.x));
  cam.y = Math.max(0, Math.min(WH-H, cam.y));
  mw.x = mouse.x + cam.x; mw.y = mouse.y + cam.y;
}

function updPlayer(dt) {
  let dx=0, dy=0;
  if (keys['w']||keys['arrowup']) dy-=1;
  if (keys['s']||keys['arrowdown']) dy+=1;
  if (keys['a']||keys['arrowleft']) dx-=1;
  if (keys['d']||keys['arrowright']) dx+=1;
  if (tl.active) { const md=60; dx+=Math.max(-1,Math.min(1,tl.dx/md)); dy+=Math.max(-1,Math.min(1,tl.dy/md)); }
  if (dx!==0||dy!==0) {
    const len=Math.sqrt(dx*dx+dy*dy);
    player.x+=(dx/len)*player.spd*dt; player.y+=(dy/len)*player.spd*dt; pushOut(player);
  }
  player.x=Math.max(WT+player.w/2,Math.min(WW-WT-player.w/2,player.x));
  player.y=Math.max(WT+player.h/2,Math.min(WH-WT-player.h/2,player.y));
  player.angle=Math.atan2(mw.y-player.y, mw.x-player.x);
  player.fireCd-=dt*1000; player.hitFlash=Math.max(0,player.hitFlash-dt*5); player.inv=Math.max(0,player.inv-dt);
  if ((mouse.down||tr.active) && player.fireCd<=0) { fire(player.x,player.y,player.angle,false); player.fireCd=player.fireRate; }
}

function updArms(dt) {
  if (player.armCount===0) return;
  for (let i=0; i<2; i++) { player.armAngles[i]+=1.4*dt; player.armCds[i]=Math.max(0,player.armCds[i]-dt*1000); }
  let target=null, minD=9999;
  const all=[...enemies]; if (boss&&!boss.entering) all.push(boss);
  all.forEach(e=>{const d=Math.hypot(e.x-player.x,e.y-player.y); if(d<minD){minD=d;target=e;}});
  for (let i=0; i<player.armCount; i++) {
    if (player.armCds[i]<=0 && target) {
      const ax=player.x+Math.cos(player.armAngles[i])*26, ay=player.y+Math.sin(player.armAngles[i])*26;
      fire(ax,ay,Math.atan2(target.y-ay,target.x-ax),true); player.armCds[i]=620;
    }
  }
}

function updBullets(dt) {
  for (let i=bullets.length-1; i>=0; i--) {
    const b=bullets[i]; b.x+=b.vx*dt; b.y+=b.vy*dt; b.life-=dt;
    if (b.life<=0||b.x<WT||b.x>WW-WT||b.y<WT||b.y>WH-WT) { bullets.splice(i,1); continue; }
    if (bulletHitsMap(b.x,b.y,true)) { bullets.splice(i,1); }
  }
}

function updEBullets(dt) {
  for (let i=enemyBullets.length-1; i>=0; i--) {
    const b=enemyBullets[i]; b.x+=b.vx*dt; b.y+=b.vy*dt; b.life-=dt;
    if (b.life<=0||b.x<WT||b.x>WW-WT||b.y<WT||b.y>WH-WT) { enemyBullets.splice(i,1); continue; }
    if (bulletHitsMap(b.x,b.y,false)) { enemyBullets.splice(i,1); continue; }
    if (player.inv<=0 && crOvlp(b.x,b.y,b.r,player.x,player.y,player.w,player.h)) {
      dmgPlayer(b.dmg); player.inv=0.1; spawnPfx(b.x,b.y,5,'#4488ff'); enemyBullets.splice(i,1);
      if (state==='GAMEOVER') return;
    }
  }
}

function updSpawn(dt) {
  if (isBossWave(wave)) return;
  spawnTimer+=dt*1000;
  if (toSpawn>0 && spawnTimer>=spawnIv) { spawnEnemy(); toSpawn--; spawnTimer=0; }
}

function updEnemies(dt) {
  for (let i=enemies.length-1; i>=0; i--) {
    const e=enemies[i];
    e.anim+=dt*4; e.hurtFlash=Math.max(0,e.hurtFlash-dt*8);
    const edx=player.x-e.x, edy=player.y-e.y, dist=Math.sqrt(edx*edx+edy*edy);

    if (e.type==='fast') {
      e.zigTimer-=dt; if(e.zigTimer<=0){e.zigDir*=-1;e.zigTimer=0.22+Math.random()*0.28;}
      if (dist>2) {
        const ba=Math.atan2(edy,edx), ma=ba+e.zigDir*0.75;
        e.x+=Math.cos(ma)*e.spd*60*dt; e.y+=Math.sin(ma)*e.spd*60*dt; pushOut(e);
      }
    } else if (e.type==='ranged') {
      const ar=220;
      if (dist>ar) { if(dist>2){e.x+=(edx/dist)*e.spd*60*dt;e.y+=(edy/dist)*e.spd*60*dt;pushOut(e);} }
      else { e.shootTimer-=dt; if(e.shootTimer<=0){fireEBullet(e.x,e.y,Math.atan2(edy,edx));e.shootTimer=2.0-Math.min(1.2,wave*0.07);} }
    } else if (e.type==='shield') {
      if (dist>2) { e.x+=(edx/dist)*e.spd*60*dt; e.y+=(edy/dist)*e.spd*60*dt; pushOut(e); }
      e.shieldAngle=Math.atan2(edy,edx);
    } else {
      if (dist>2) { e.x+=(edx/dist)*e.spd*60*dt; e.y+=(edy/dist)*e.spd*60*dt; pushOut(e); }
    }

    if (player.inv<=0 && rrOvlp(player.x,player.y,player.w+2,player.h+2,e.x,e.y,e.w,e.h)) {
      e.dmgTimer+=dt*1000;
      if (e.dmgTimer>=700) { dmgPlayer(e.dmg); player.inv=0.12; e.dmgTimer=0; if(state==='GAMEOVER')return; }
    }

    for (let j=bullets.length-1; j>=0; j--) {
      const b=bullets[j];
      if (!rrOvlp(b.x,b.y,b.w,b.h,e.x,e.y,e.w,e.h)) continue;
      if (e.type==='shield') {
        if (angDiff(Math.atan2(b.vy,b.vx), e.shieldAngle+Math.PI) < Math.PI*0.55) {
          spawnPfx(b.x,b.y,7,'#ffaa00'); bullets.splice(j,1); break;
        }
      }
      e.hp-=25; e.hurtFlash=1;
      spawnPfx(b.x,b.y,4,e.type==='heavy'?'#ff8800':'#ff3366'); bullets.splice(j,1);
      if (e.hp<=0) {
        score+=e.pts; coins+=e.cn;
        if(AC) sndDie(e.type);
        spawnPfx(e.x,e.y,e.type==='heavy'?16:10,e.type==='heavy'?'#ff8800':'#cc44ff');
        spawnTxt(e.x,e.y-10,'+'+e.pts,'#ffffff'); spawnTxt(e.x,e.y-26,'+'+e.cn+'c','#ffdd00');
        enemies.splice(i,1); waveLeft--;
        if (waveLeft<=0&&toSpawn<=0&&!boss){state='WAVE_TRANSITION';transTimer=1.5;}
      }
      break;
    }
  }
}

function updBoss(dt) {
  if (!boss) return;
  boss.anim+=dt*3; boss.hurtFlash=Math.max(0,boss.hurtFlash-dt*6);
  if (boss.entering) { boss.y+=90*dt; if(boss.y>=boss.targetY){boss.y=boss.targetY;boss.entering=false;} return; }
  boss.phase=boss.hp<boss.maxHp*0.5?1:0;
  boss.oAngle+=dt*(boss.phase===1?0.7:0.45);
  boss.oRadius=Math.min(boss.oRadius+dt*35,boss.oRadiusTgt);
  boss.x=WW/2+Math.cos(boss.oAngle)*boss.oRadius;
  boss.y=boss.targetY+Math.sin(boss.oAngle*0.8)*28;
  boss.atkTimer+=dt;
  if (boss.atkTimer>=(boss.phase===1?boss.atkCd*0.65:boss.atkCd)) {
    boss.atkTimer=0;
    if(boss.atkIdx%3===0)bossRadial(); else if(boss.atkIdx%3===1)bossShockwave(); else bossSlam();
    boss.atkIdx++;
  }
  if (player.inv<=0&&rrOvlp(player.x,player.y,player.w+4,player.h+4,boss.x,boss.y,boss.w,boss.h)) {
    dmgPlayer(25); player.inv=0.3; if(state==='GAMEOVER')return;
  }
  for (let j=bullets.length-1; j>=0; j--) {
    const b=bullets[j];
    if (!rrOvlp(b.x,b.y,b.w,b.h,boss.x,boss.y,boss.w,boss.h)) continue;
    boss.hp-=b.fromArm?20:25; boss.hurtFlash=1; spawnPfx(b.x,b.y,5,'#ff6600'); bullets.splice(j,1);
    if (boss.hp<=0) {
      score+=500; coins+=150;
      spawnPfx(boss.x,boss.y,40,'#ff8800'); spawnPfx(boss.x,boss.y,30,'#ffff00');
      spawnTxt(boss.x,boss.y-20,'BOSS! +500','#ffdd00'); spawnTxt(boss.x,boss.y-38,'+150 coin','#ffdd00');
      boss=null; bossProjs=[]; aoeZones=[]; shockwaves=[];
      waveLeft=0; state='WAVE_TRANSITION'; transTimer=1.5;
    }
    break;
  }
}

function updBossProjs(dt) {
  for (let i=bossProjs.length-1; i>=0; i--) {
    const p=bossProjs[i]; p.x+=p.vx*dt; p.y+=p.vy*dt; p.life-=dt;
    if (p.life<=0||bulletHitsMap(p.x,p.y,false)){bossProjs.splice(i,1);continue;}
    if (player.inv<=0&&crOvlp(p.x,p.y,p.r,player.x,player.y,player.w,player.h)) {
      dmgPlayer(p.dmg); player.inv=0.1; spawnPfx(p.x,p.y,6,'#ff4400'); bossProjs.splice(i,1);
      if(state==='GAMEOVER')return;
    }
  }
}

function updAoe(dt) {
  for (let i=aoeZones.length-1; i>=0; i--) {
    const z=aoeZones[i];
    if (!z.exploded) {
      z.timer-=dt;
      if (z.timer<=0) {
        z.exploded=true; z.flashTimer=0.45;
        if (Math.hypot(player.x-z.x,player.y-z.y)<z.r&&player.inv<=0){dmgPlayer(z.dmg);player.inv=0.2;if(state==='GAMEOVER')return;}
        spawnPfx(z.x,z.y,22,'#ff4400');
      }
    } else { z.flashTimer-=dt; if(z.flashTimer<=0)aoeZones.splice(i,1); }
  }
}

function updShock(dt) {
  for (let i=shockwaves.length-1; i>=0; i--) {
    const s=shockwaves[i]; s.r+=s.spd*dt; s.alpha=1-s.r/s.maxR;
    if (s.r>=s.maxR||s.alpha<=0){shockwaves.splice(i,1);continue;}
    if (!s.hit&&player.inv<=0) {
      const d=Math.hypot(player.x-s.x,player.y-s.y);
      if (d>=s.r-18&&d<=s.r+18){s.hit=true;dmgPlayer(s.dmg);player.inv=0.15;if(state==='GAMEOVER')return;}
    }
  }
}

function updPfx(dt) {
  for (let i=particles.length-1; i>=0; i--) {
    const p=particles[i]; p.x+=p.vx*dt; p.y+=p.vy*dt; p.vx*=0.92; p.vy*=0.92;
    p.life-=dt; p.alpha=Math.max(0,p.life/p.maxLife); if(p.life<=0)particles.splice(i,1);
  }
}
function updFtxt(dt) {
  for (let i=floatTexts.length-1; i>=0; i--) {
    const t=floatTexts[i]; t.y+=t.vy*dt; t.life-=dt; t.alpha=Math.max(0,t.life/t.maxLife);
    if(t.life<=0)floatTexts.splice(i,1);
  }
}

// ── DRAW ───────────────────────────────────────────────────────────
function drawWorld() {
  const gColors = { grass:'#0d1f0d', stone:'#272725', metal:'#1a1a2a' };
  const gGrid   = { grass:'#122012', stone:'#2e2e2b', metal:'#1f1f32' };
  gndZones.forEach(z => {
    ctx.fillStyle = gColors[z.type] || '#0d1f0d';
    ctx.fillRect(z.x, z.y, z.w, z.h);
    ctx.strokeStyle = gGrid[z.type] || '#122012'; ctx.lineWidth = 1;
    for (let x = z.x; x <= z.x+z.w; x+=40) { ctx.beginPath(); ctx.moveTo(x,z.y); ctx.lineTo(x,z.y+z.h); ctx.stroke(); }
    for (let y = z.y; y <= z.y+z.h; y+=40) { ctx.beginPath(); ctx.moveTo(z.x,y); ctx.lineTo(z.x+z.w,y); ctx.stroke(); }
  });
  // Out-of-zone fill (border area)
  ctx.fillStyle = '#080e08';
  ctx.fillRect(0,0,WW,WT); ctx.fillRect(0,WH-WT,WW,WT);
  ctx.fillRect(0,0,WT,WH); ctx.fillRect(WW-WT,0,WT,WH);
}

function drawMapObjs() {
  mapObjs.forEach(o => {
    if (!o.solid && o.type!=='crate') return; // skip destroyed non-crates... wait, destroyed crates have solid=false, skip them
    if (o.type === 'crate' && !o.solid) return;
    if (o.type === 'wall') {
      ctx.fillStyle = '#2a2a2a'; ctx.fillRect(o.x, o.y, o.w, o.h);
      ctx.fillStyle = '#333333'; ctx.fillRect(o.x+2, o.y+2, o.w-4, o.h-4);
      ctx.fillStyle = '#1a1a1a'; ctx.fillRect(o.x, o.y, o.w, 3); ctx.fillRect(o.x, o.y, 3, o.h);
    } else if (o.type === 'sandbag') {
      ctx.fillStyle = '#8a7040'; ctx.fillRect(o.x, o.y, o.w, o.h);
      ctx.fillStyle = '#a0864e'; ctx.fillRect(o.x+2, o.y+2, o.w-4, o.h-4);
      ctx.fillStyle = '#6a5530'; ctx.fillRect(o.x, o.y+o.h/2, o.w, 2);
      ctx.fillStyle = '#c8a870'; ctx.fillRect(o.x+4, o.y+3, o.w-8, 3);
    } else if (o.type === 'crate') {
      const hp = o.hp/o.maxHp;
      ctx.fillStyle = hp>0.5?'#6a4a20':'#4a3015'; ctx.fillRect(o.x, o.y, o.w, o.h);
      ctx.fillStyle = hp>0.5?'#8a6030':'#5a3a1a'; ctx.fillRect(o.x+3, o.y+3, o.w-6, o.h-6);
      ctx.fillStyle = '#2a1a08';
      ctx.fillRect(o.x+o.w/2-1, o.y+2, 2, o.h-4); ctx.fillRect(o.x+2, o.y+o.h/2-1, o.w-4, 2);
      ctx.strokeStyle = '#3a2810'; ctx.lineWidth=1; ctx.strokeRect(o.x,o.y,o.w,o.h);
    } else if (o.type === 'fence') {
      ctx.fillStyle = '#556655'; ctx.fillRect(o.x, o.y, o.w, o.h);
      ctx.fillStyle = '#445544';
      if (o.w > o.h) { for(let x=o.x; x<o.x+o.w; x+=16) ctx.fillRect(x, o.y, 4, o.h); }
      else           { for(let y=o.y; y<o.y+o.h; y+=16) ctx.fillRect(o.x, y, o.w, 4); }
    }
  });
}

function drawEnemy(e) {
  ctx.save(); ctx.translate(e.x, e.y);
  const bob = Math.sin(e.anim)*1.5;
  if (e.hurtFlash>0) {
    ctx.globalAlpha=e.hurtFlash*0.8; ctx.fillStyle='#fff';
    ctx.fillRect(-e.w/2-2,-e.h/2-2+bob,e.w+4,e.h+4); ctx.globalAlpha=1;
  }

  if (e.type==='basic') {
    ctx.fillStyle='#5a1a6a'; ctx.fillRect(-7,-8+bob,14,16);
    ctx.fillStyle='#7a2a8a'; ctx.fillRect(-6,-7+bob,12,7); ctx.fillRect(-5,1+bob,10,7);
    ctx.fillStyle='#9a3aaa'; ctx.fillRect(-5,-8+bob,10,5);
    ctx.fillStyle='#ffccff'; ctx.fillRect(-3,-6+bob,2,2); ctx.fillRect(2,-6+bob,2,2);

function drawBossEffects() {
  aoeZones.forEach(z=>{
    ctx.save();
    if (!z.exploded) {
      const pulse=Math.sin((1-z.timer/z.warnDur)*Math.PI*9)*0.5+0.5;
      ctx.globalAlpha=0.22+pulse*0.42; ctx.fillStyle='#ff3300';
      ctx.beginPath(); ctx.arc(z.x,z.y,z.r,0,Math.PI*2); ctx.fill();
      ctx.globalAlpha=0.8; ctx.strokeStyle='#ff0000'; ctx.lineWidth=2;
      ctx.beginPath(); ctx.arc(z.x,z.y,z.r,0,Math.PI*2); ctx.stroke();
    } else {
      ctx.globalAlpha=z.flashTimer/0.45; ctx.fillStyle='#ff8800';
      ctx.beginPath(); ctx.arc(z.x,z.y,z.r*1.3,0,Math.PI*2); ctx.fill();
    }
    ctx.restore();
  });
  shockwaves.forEach(s=>{
    ctx.save(); ctx.globalAlpha=s.alpha*0.8; ctx.strokeStyle='#ff3300'; ctx.lineWidth=5;
    ctx.beginPath(); ctx.arc(s.x,s.y,s.r,0,Math.PI*2); ctx.stroke();
    ctx.globalAlpha=s.alpha*0.3; ctx.strokeStyle='#ff9900'; ctx.lineWidth=10;
    ctx.beginPath(); ctx.arc(s.x,s.y,Math.max(0,s.r-5),0,Math.PI*2); ctx.stroke();
    ctx.restore();
  });
  bossProjs.forEach(p=>{
    ctx.save(); ctx.globalAlpha=Math.min(1,p.life*2); ctx.fillStyle=p.color;
    ctx.beginPath(); ctx.arc(p.x,p.y,p.r,0,Math.PI*2); ctx.fill();
    ctx.fillStyle='#ffccaa'; ctx.beginPath(); ctx.arc(p.x,p.y,p.r*0.45,0,Math.PI*2); ctx.fill();
    ctx.restore();
  });
}

function drawLighting() {
  const sx = player.x - cam.x, sy = player.y - cam.y;
  const grad = ctx.createRadialGradient(sx,sy,55,sx,sy,360);
  grad.addColorStop(0,'rgba(0,0,0,0)');
  grad.addColorStop(0.55,'rgba(0,0,0,0.15)');

function drawMinimap() {
  const mw_=120, mh=90, mx=W-mw_-10, my=10, scale=mw_/WW;
  ctx.save();
  ctx.fillStyle='rgba(0,0,0,0.6)'; ctx.fillRect(mx,my,mw_,mh);
  ctx.strokeStyle='#444'; ctx.lineWidth=1; ctx.strokeRect(mx,my,mw_,mh);
  mapObjs.forEach(o=>{
    if (!o.solid) return;
    if(o.type==='wall'){ctx.fillStyle='#444';}
    else if(o.type==='sandbag'){ctx.fillStyle='#665530';}
    else if(o.type==='crate'){ctx.fillStyle='#553310';}
    else{ctx.fillStyle='#334433';}
    ctx.fillRect(mx+o.x*scale, my+o.y*(mh/WH), Math.max(1,o.w*scale), Math.max(1,o.h*(mh/WH)));
  });
  enemies.forEach(e=>{
    const col={basic:'#cc44ff',fast:'#88ff44',ranged:'#4488ff',shield:'#ffaa44',heavy:'#ff4444'}[e.type]||'#cc44ff';
    ctx.fillStyle=col; ctx.fillRect(mx+e.x*scale-1,my+e.y*(mh/WH)-1,3,3);
  });
  if (boss) { ctx.fillStyle='#ff0000'; ctx.fillRect(mx+boss.x*scale-2,my+boss.y*(mh/WH)-2,5,5); }
  ctx.fillStyle='#00ff88'; ctx.fillRect(mx+player.x*scale-2,my+player.y*(mh/WH)-2,4,4);
  const vx=mx+cam.x*scale, vy=my+cam.y*(mh/WH), vw=W*scale, vh=H*(mh/WH);
  ctx.strokeStyle='rgba(255,255,255,0.4)'; ctx.lineWidth=1; ctx.strokeRect(vx,vy,vw,vh);
  ctx.restore();
}

  ctx.fillStyle='#1a6622'; ctx.fillRect(bx,by,220,48);
  ctx.strokeStyle='#44ee44'; ctx.lineWidth=2; ctx.strokeRect(bx,by,220,48);
  ctx.fillStyle='#44ff44'; ctx.font='bold 16px monospace'; ctx.fillText('TEKRAR OYNA  [R]', W/2, by+30);
  ctx.restore();
}

// ── RENDER ─────────────────────────────────────────────────────────
function render() {
  ctx.fillStyle='#080e08'; ctx.fillRect(0,0,W,H);

  if (state==='SHOP') {
    ctx.save(); ctx.translate(Math.round(-cam.x),Math.round(-cam.y));
    drawWorld(); drawMapObjs(); ctx.restore();
    drawShop(); return;
  }
  if (state==='GAMEOVER') {
    ctx.save(); ctx.translate(Math.round(-cam.x),Math.round(-cam.y));
    drawWorld(); drawMapObjs(); drawParticles(); ctx.restore();
    drawGameOver(); return;
  }

  ctx.save(); ctx.translate(Math.round(-cam.x),Math.round(-cam.y));
  drawWorld(); drawMapObjs();
  drawBossEffects(); drawParticles();
  drawBullets(); enemies.forEach(drawEnemy);
  drawBoss(); drawPlayer(); drawFloatTexts();
  ctx.restore();

  drawLighting(); drawHUD();
}

// ── LOOP ───────────────────────────────────────────────────────────
function loop(ts) {
  const dt = Math.min((ts - lastTime) / 1000, 0.05);
  lastTime = ts; update(dt); render();
  requestAnimationFrame(loop);
}
requestAnimationFrame(ts => { lastTime = ts; requestAnimationFrame(loop); });
